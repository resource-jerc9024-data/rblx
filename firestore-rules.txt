rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper function to check if user is admin via Firebase Auth
    function isAdmin() {
      // Get admin emails - REPLACE with actual admin emails
      let adminEmails = ['pratiksrisantoppo@gmail.com', 'test@admin.com', 'your-admin@gmail.com', 'admin@yourdomain.com'];
      return request.auth != null && request.auth.token.email in adminEmails;
    }
    
    // Helper function to check if user owns document
    function isOwner(userId) {
      return request.auth != null && 
             request.auth.uid == userId ||
             (resource != null && resource.data.userId == request.auth.uid);
    }
    
    // Users collection
    match /users/{userId} {
      // Allow read access to all users (for leaderboard)
      allow read: if true;
      
      // Allow anyone to create users (for Roblox OAuth registration)
      allow create: if request.resource.data.keys().hasAll(['userId', 'username']) &&
                      request.resource.data.userId is string &&
                      request.resource.data.username is string &&
                      request.resource.data.username.size() < 50;
      
      // Users can update their own profile, admins can update any profile
      allow update: if isOwner(userId) || isAdmin();
      
      // Only admins can delete users
      allow delete: if isAdmin();
      
      // Validate user data structure
      allow write: if request.resource.data.keys().hasAll(['userId', 'username']) &&
                      request.resource.data.keys().hasAny([
                        'userId', 'username', 'displayName', 'avatarUrl', 
                        'lastLogin', 'createdVia', 'createdAt', 'badges'
                      ]) &&
                      request.resource.data.userId is string &&
                      request.resource.data.username is string &&
                      request.resource.data.username.size() > 0 &&
                      request.resource.data.username.size() < 50;
    }
    
    // Transactions collection
    match /transactions/{transactionId} {
      // Anyone can read transactions (for leaderboard)
      allow read: if true;
      
      // Allow anyone to create transactions (for donation system)
      allow create: if request.resource.data.keys().hasAll(['userId', 'username', 'amount']) &&
                      request.resource.data.userId is string &&
                      request.resource.data.username is string &&
                      request.resource.data.amount is number &&
                      request.resource.data.amount > 0 &&
                      request.resource.data.amount < 1000000 && // Max donation amount
                      request.resource.data.timestamp is timestamp;
      
      // Only admins can update/delete transactions
      allow update, delete: if isAdmin();
      
      // Validate transaction data structure
      allow write: if request.resource.data.keys().hasAny([
        'userId', 'username', 'amount', 'note', 
        'timestamp', 'month', 'year'
      ]) &&
                      request.resource.data.userId is string &&
                      request.resource.data.username is string &&
                      request.resource.data.amount is number &&
                      request.resource.data.amount > 0 &&
                      request.resource.data.amount < 1000000 &&
                      request.resource.data.timestamp is timestamp;
    }
    
    // Admin configuration collection (for admin settings)
    match /admin/{configId} {
      // Only admins can read/write admin config
      allow read, write: if isAdmin();
    }
    
    // Badges collection (for user achievement caching)
    match /badges/{userId} {
      // Users can read their own badges, admins can read all
      allow read: if request.auth != null && 
                 (request.auth.uid == userId || isAdmin());
      
      // Users can update their own badges, admins can update any
      allow write: if request.auth != null && 
                  (request.auth.uid == userId || isAdmin());
    }
  }
}
